18 /*
19  * General State-Space Controller class.
20  *
21  * See "example_StateSpaceController.cpp.example" to have an implementation example.
22  *
23  * StateSpaceController class
24  * ______________/\______________
25  * / ___ ________ \ ________
26  * r +/ \ e | | u | | y
27  * ---->| |---->| K |---->| Plant |---+-->
28  * \___/ |________| |________| |
29  * ^ - |
30  * y |____________________________________|
31  *
32  * K:
33  * | x_{i+1} = A*x_i + B*e_i
34  * | u_i = C*x_i + D*e_i
35  *
36  * with e_i = r_i - y_i
37  */
